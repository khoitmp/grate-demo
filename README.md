## Command

  --connectionstring="User ID=postgres;Password=Pass123!;Host=localhost;Port=5432;Database=core;Pooling=true" \
  --sqlfilesdirectory=sql \
  --databasetype=PostgreSQL \
  --environment=Local|TH|ID
